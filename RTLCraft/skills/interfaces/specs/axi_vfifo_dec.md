- `SEG_WIDTH = 32`
- `SEG_CNT = 2`
- `AXIS_DATA_WIDTH = SEG_WIDTH*SEG_CNT/2`
- `AXIS_KEEP_ENABLE = (AXIS_DATA_WIDTH>8)`
- `AXIS_KEEP_WIDTH = (AXIS_DATA_WIDTH/8)`
- `AXIS_LAST_ENABLE = 1`
- `AXIS_ID_ENABLE = 0`
- `AXIS_ID_WIDTH = 8`
- `AXIS_DEST_ENABLE = 0`
- `AXIS_DEST_WIDTH = 8`
- `AXIS_USER_ENABLE = 1`
- `AXIS_USER_WIDTH = 1`
- `AXIS_KEEP_WIDTH_INT = AXIS_KEEP_ENABLE ? AXIS_KEEP_WIDTH : 1`
- `AXIS_BYTE_LANES = AXIS_KEEP_WIDTH_INT`
- `AXIS_BYTE_SIZE = AXIS_DATA_WIDTH/AXIS_BYTE_LANES`
- `AXIS_BYTE_IDX_WIDTH = $clog2(AXIS_BYTE_LANES)`
- `BYTE_SIZE = AXIS_BYTE_SIZE`
- `SEG_BYTE_LANES = SEG_WIDTH / BYTE_SIZE`
- `EXPAND_INPUT = SEG_CNT < 2`
- `SEG_CNT_INT = EXPAND_INPUT ? SEG_CNT*2 : SEG_CNT`
- `SEG_IDX_WIDTH = $clog2(SEG_CNT_INT)`
- `SEG_BYTE_IDX_WIDTH = $clog2(SEG_BYTE_LANES)`
- `AXIS_SEG_CNT = (AXIS_DATA_WIDTH + SEG_WIDTH-1) / SEG_WIDTH`
- `AXIS_SEG_IDX_WIDTH = AXIS_SEG_CNT > 1 ? $clog2(AXIS_SEG_CNT) : 1`
- `AXIS_LEN_MASK = AXIS_BYTE_LANES-1`
- `OUT_OFFS_WIDTH = AXIS_SEG_IDX_WIDTH`
- `META_ID_OFFSET = 0`
- `META_DEST_OFFSET = META_ID_OFFSET + (AXIS_ID_ENABLE ? AXIS_ID_WIDTH : 0)`
- `META_USER_OFFSET = META_DEST_OFFSET + (AXIS_DEST_ENABLE ? AXIS_DEST_WIDTH : 0)`
- `META_WIDTH = META_USER_OFFSET + (AXIS_USER_ENABLE ? AXIS_USER_WIDTH : 0)`
- `HDR_SIZE = (16 + META_WIDTH + BYTE_SIZE-1) / BYTE_SIZE`
- `HDR_WIDTH = HDR_SIZE * BYTE_SIZE`
- `HDR_LEN_WIDTH = 12`
- `HDR_SEG_LEN_WIDTH = HDR_LEN_WIDTH-SEG_BYTE_IDX_WIDTH`
- `CTRL_FIFO_ADDR_WIDTH = 5`
- `OUTPUT_FIFO_ADDR_WIDTH = 5`
- `CTRL_FIFO_PTR_WIDTH = CTRL_FIFO_ADDR_WIDTH + SEG_IDX_WIDTH`
